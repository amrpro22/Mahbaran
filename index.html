  // دالة بسيطة لتجنب XSS

  function escapeHtml(str) {

    if (!str) return '';

    return str.replace(/[&<>]/g, function(m) {

      if (m === '&') return '&amp;';

      if (m === '<') return '&lt;';

      if (m === '>') return '&gt;';

      return m;

    });

  }

    const inputElement = document.getElementById("searchNameInput");

    const query = inputElement.value;
